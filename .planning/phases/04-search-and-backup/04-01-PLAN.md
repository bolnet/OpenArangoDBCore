#pragma once
#ifndef ARANGODB_MIN_HASH_FUNCTIONS_H
#define ARANGODB_MIN_HASH_FUNCTIONS_H

#include <cstdint>
#include <string>
#include <string_view>
#include <vector>

namespace arangodb {

/// Default number of hash functions for MinHash signatures.
static constexpr uint32_t kDefaultMinHashK = 128;

/// Large Mersenne prime for modular arithmetic in hash family.
/// p = 2^61 - 1 (largest Mersenne prime fitting in uint64).
static constexpr uint64_t kMinHashPrime = (1ULL << 61) - 1;

/// A single permutation in the universal hash family: h(x) = (a*x + b) mod p.
struct PermutationSeed {
  uint64_t a;  // multiplier (must be in [1, p-1])
  uint64_t b;  // offset (must be in [0, p-1])
};

/// Precomputed set of k permutation seeds for MinHash computation.
/// Seeds are deterministically generated from a master seed so that
/// all nodes in a cluster produce identical signatures for the same input.
std::vector<PermutationSeed> generatePermutationSeeds(
    uint32_t k, uint64_t masterSeed = 0x9E3779B97F4A7C15ULL);

/// MinHashGenerator computes MinHash signatures over arbitrary string sets.
///
/// Usage:
///   auto seeds = generatePermutationSeeds(128);
///   MinHashGenerator gen(seeds);
///   gen.addElement("foo");
///   gen.addElement("bar");
///   auto sig = gen.finalize();  // vector of 128 uint64 values
class MinHashGenerator {
 public:
  /// Construct with precomputed permutation seeds.
  /// Does NOT take ownership -- caller must keep seeds alive.
  explicit MinHashGenerator(std::vector<PermutationSeed> const& seeds);

  /// Hash a single set element and update running minimums.
  void addElement(std::string_view element);

  /// Return the final signature (vector of k minimum hash values).
  /// After calling finalize(), the generator is consumed; do not reuse.
  std::vector<uint64_t> finalize() const;

  /// Reset the generator to compute a new signature with the same seeds.
  void reset();

  /// Number of hash functions (k).
  uint32_t k() const { return static_cast<uint32_t>(_minimums.size()); }

 private:
  std::vector<PermutationSeed> const& _seeds;
  std::vector<uint64_t> _minimums;  // running min for each hash function
};

/// Compute estimated Jaccard similarity from two MinHash signatures.
/// Returns value in [0.0, 1.0]. Signatures must have equal length.
double estimateJaccard(std::vector<uint64_t> const& sig1,
                       std::vector<uint64_t> const& sig2);

// ---- AQL Function Registration ----

/// Register MINHASH and MINHASH_MATCH with the AQL function registry.
/// Called from AqlFunctionFeature::addEnterpriseFunctions().
void registerMinHashAqlFunctions();

}  // namespace arangodb

#endif  // ARANGODB_MIN_HASH_FUNCTIONS_H

#include "MinHashFunctions.h"
#include <algorithm>
#include <cstring>
#include <limits>
#include <stdexcept>

// Use xxHash for base hashing of set elements.
// Falls back to a built-in FNV-1a if xxHash header is unavailable.
#if __has_include("xxhash.h")
#include "xxhash.h"
#define MINHASH_USE_XXHASH 1
#else
#define MINHASH_USE_XXHASH 0
#endif

namespace arangodb {

namespace {

/// Base hash: convert a string element to a uint64 using xxHash or FNV-1a.
uint64_t baseHash(std::string_view element, uint64_t seed) {
#if MINHASH_USE_XXHASH
  return XXH64(element.data(), element.size(), seed);
#else
  // FNV-1a 64-bit fallback
  uint64_t hash = 14695981039346656037ULL ^ seed;
  for (char c : element) {
    hash ^= static_cast<uint64_t>(static_cast<unsigned char>(c));
    hash *= 1099511628211ULL;
  }
  return hash;
#endif
}

/// Modular multiplication: (a * b) mod p where p = 2^61 - 1.
/// Uses __uint128_t to avoid overflow.
uint64_t mulmod(uint64_t a, uint64_t b, uint64_t p) {
  __uint128_t result = static_cast<__uint128_t>(a) * b;
  // Fast reduction for Mersenne prime 2^61 - 1
  uint64_t lo = static_cast<uint64_t>(result) & p;
  uint64_t hi = static_cast<uint64_t>(result >> 61);
  uint64_t val = lo + hi;
  return val >= p ? val - p : val;
}

/// Apply permutation hash: h(x) = (a*x + b) mod p.
uint64_t permutationHash(uint64_t x, PermutationSeed const& seed) {
  uint64_t p = kMinHashPrime;
  uint64_t ax = mulmod(seed.a, x, p);
  uint64_t result = ax + seed.b;
  return result >= p ? result - p : result;
}

}  // namespace

// --- generatePermutationSeeds ---

std::vector<PermutationSeed> generatePermutationSeeds(
    uint32_t k, uint64_t masterSeed) {
  std::vector<PermutationSeed> seeds;
  seeds.reserve(k);
  for (uint32_t i = 0; i < k; ++i) {
    // Derive deterministic a, b from masterSeed and index.
    uint64_t a = baseHash(std::string_view(
        reinterpret_cast<char const*>(&i), sizeof(i)), masterSeed);
    uint64_t b = baseHash(std::string_view(
        reinterpret_cast<char const*>(&i), sizeof(i)), masterSeed ^ 0xDEADBEEF);
    // Ensure a is in [1, p-1] and b is in [0, p-1]
    a = (a % (kMinHashPrime - 1)) + 1;
    b = b % kMinHashPrime;
    seeds.push_back({a, b});
  }
  return seeds;
}

// --- MinHashGenerator ---

MinHashGenerator::MinHashGenerator(
    std::vector<PermutationSeed> const& seeds)
    : _seeds(seeds),
      _minimums(seeds.size(), std::numeric_limits<uint64_t>::max()) {}

void MinHashGenerator::addElement(std::string_view element) {
  // Base-hash the element once.
  uint64_t h = baseHash(element, 0);
  // Apply each permutation and track minimums.
  for (size_t i = 0; i < _seeds.size(); ++i) {
    uint64_t ph = permutationHash(h, _seeds[i]);
    if (ph < _minimums[i]) {
      _minimums[i] = ph;
    }
  }
}

std::vector<uint64_t> MinHashGenerator::finalize() const {
  return _minimums;
}

void MinHashGenerator::reset() {
  std::fill(_minimums.begin(), _minimums.end(),
            std::numeric_limits<uint64_t>::max());
}

// --- estimateJaccard ---

double estimateJaccard(std::vector<uint64_t> const& sig1,
                       std::vector<uint64_t> const& sig2) {
  if (sig1.size() != sig2.size() || sig1.empty()) {
    return 0.0;
  }
  uint32_t matches = 0;
  for (size_t i = 0; i < sig1.size(); ++i) {
    if (sig1[i] == sig2[i]) {
      ++matches;
    }
  }
  return static_cast<double>(matches) / static_cast<double>(sig1.size());
}

void registerMinHashAqlFunctions() {
  // Registration happens in Task 3 after AQL value layer is implemented.
}

}  // namespace arangodb

// --- aqlMinHash ---

AqlValue aqlMinHash(ExpressionContext* /*ctx*/, AstNode const& /*node*/,
                    VPackFunctionParametersView args) {
  // 1. Validate first arg is array
  VPackSlice setSlice = args[0].slice();
  if (!setSlice.isArray()) {
    return AqlValue(AqlValueHintNull{});
  }

  // 2. Parse k (default 128)
  uint32_t k = kDefaultMinHashK;
  if (args.size() > 1) {
    VPackSlice kSlice = args[1].slice();
    if (kSlice.isNumber()) {
      int64_t kVal = kSlice.getNumber<int64_t>();
      if (kVal <= 0 || kVal > 4096) {
        return AqlValue(AqlValueHintNull{});
      }
      k = static_cast<uint32_t>(kVal);
    } else if (!kSlice.isNull() && !kSlice.isNone()) {
      return AqlValue(AqlValueHintNull{});
    }
  }

  // 3. Generate seeds and compute signature
  auto seeds = generatePermutationSeeds(k);
  MinHashGenerator gen(seeds);
  for (VPackArrayIterator it(setSlice); it.valid(); ++it) {
    VPackSlice elem = *it;
    // Coerce element to string representation for hashing
    if (elem.isString()) {
      gen.addElement(elem.stringView());
    } else {
      // Use VPack JSON serialization as string fallback
      std::string s = elem.toJson();
      gen.addElement(s);
    }
  }

  // 4. Build result array
  auto sig = gen.finalize();
  VPackBuilder builder;
  builder.openArray();
  for (uint64_t v : sig) {
    builder.add(VPackValue(v));
  }
  builder.close();
  return AqlValue(builder);
}

// --- aqlMinHashMatch ---

AqlValue aqlMinHashMatch(ExpressionContext* /*ctx*/, AstNode const& /*node*/,
                         VPackFunctionParametersView args) {
  // 1. Validate both signatures are arrays
  VPackSlice sig1Slice = args[0].slice();
  VPackSlice sig2Slice = args[1].slice();
  if (!sig1Slice.isArray() || !sig2Slice.isArray()) {
    return AqlValue(AqlValueHintNull{});
  }

  // 2. Validate same length
  if (sig1Slice.length() != sig2Slice.length() || sig1Slice.length() == 0) {
    return AqlValue(AqlValueHintNull{});
  }

  // 3. Parse threshold
  VPackSlice threshSlice = args[2].slice();
  if (!threshSlice.isNumber()) {
    return AqlValue(AqlValueHintNull{});
  }
  double threshold = threshSlice.getNumber<double>();
  if (threshold < 0.0 || threshold > 1.0) {
    return AqlValue(AqlValueHintNull{});
  }

  // 4. Compute Jaccard estimate
  uint64_t matches = 0;
  uint64_t total = sig1Slice.length();
  VPackArrayIterator it1(sig1Slice);
  VPackArrayIterator it2(sig2Slice);
  while (it1.valid() && it2.valid()) {
    if ((*it1).getNumber<uint64_t>() == (*it2).getNumber<uint64_t>()) {
      ++matches;
    }
    ++it1;
    ++it2;
  }

  double jaccard = static_cast<double>(matches) / static_cast<double>(total);
  return AqlValue(AqlValueHintBool{}, jaccard >= threshold);
}

///   2. MinHashGenerator computes k signatures over the token set

  bool reset(std::string_view data);

  /// Returns false when all k tokens have been emitted.
  bool next();

  /// Current token value (string representation of uint64 signature component).
  std::string_view value() const;

  /// Access the full config.

bool MinHashAnalyzer::reset(std::string_view data) {
  _signatureTokens.clear();
  _currentToken = 0;

  // Step 1: Tokenize input using inner analyzer (space-delimited by default)
  std::vector<std::string> tokens;
  // Simple space tokenizer for delimiter type
  std::istringstream stream(std::string(data));
  std::string token;
  while (stream >> token) {
    tokens.push_back(std::move(token));

  MinHashGenerator gen(seeds);

  }
  auto signature = gen.finalize();

  // Step 3: Convert signature values to string tokens for indexing
  _signatureTokens.reserve(signature.size());
  for (size_t i = 0; i < signature.size(); ++i) {
    // Prefix with slot index for positional matching:
    // "slot:value" ensures slot 0 of sig1 is compared with slot 0 of sig2
    _signatureTokens.push_back(
        std::to_string(i) + ":" + std::to_string(signature[i]));
  }

  return !_signatureTokens.empty() || tokens.empty();
}

bool MinHashAnalyzer::next() {
  if (_currentToken >= _signatureTokens.size()) {
    return false;
  }
  ++_currentToken;
  return true;
}

std::string_view MinHashAnalyzer::value() const {
  if (_currentToken == 0 || _currentToken > _signatureTokens.size()) {
    return {};
  }
  return _signatureTokens[_currentToken - 1];
}

<pitfalls>
1. **Overflow in modular arithmetic**: The hash family h(x) = (a*x + b) mod p with p = 2^61-1 requires 128-bit intermediate multiplication. Use __uint128_t; do NOT attempt uint64 multiplication followed by modulus -- it will silently overflow and destroy hash independence.

2. **Degenerate permutation seeds**: If a=0 in any permutation, h(x) = b (constant), collapsing that hash slot. Always enforce a in [1, p-1].

3. **Non-deterministic signatures across nodes**: If permutation seeds use random state (e.g., std::random_device), different cluster nodes produce different signatures for the same input. Use a fixed masterSeed constant, not randomized seeds.

4. **O(k * |set|) performance on large sets**: For sets >10K elements, the inner loop iterates k times per element. With k=128 this is manageable, but if users pass k=4096 with 100K elements, that is 400M operations. Cap k at 4096 and document the complexity.

5. **Coercing non-string array elements**: If MINHASH receives an array of mixed types [1, "foo", true], each element must be deterministically converted to a string before hashing. Using VPack JSON serialization (toJson()) ensures consistency, but the caller must understand that [1] and ["1"] will produce different signatures.

6. **Signature token slot indexing in analyzer**: The MinHash analyzer must emit tokens prefixed with their slot index ("0:hash", "1:hash", ...) so IResearch matches slot 0 of one document against slot 0 of another. Without positional prefixes, any-to-any matching produces incorrect Jaccard estimates.

7. **Seed caching**: generatePermutationSeeds() allocates and computes k seeds on every call. For MINHASH() in a query scanning millions of documents, this is wasteful. Cache seeds per k value using a thread-safe map (std::shared_mutex + std::unordered_map<uint32_t, vector<PermutationSeed>>). However, do NOT prematurely optimize in the first implementation -- add caching only after profiling confirms it is a bottleneck.
</pitfalls>
