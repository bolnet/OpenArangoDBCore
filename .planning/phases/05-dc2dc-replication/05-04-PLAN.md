```cpp
#pragma once
#ifndef ARANGODB_REST_REPLICATION_HANDLER_H
#define ARANGODB_REST_REPLICATION_HANDLER_H

#include <string>
#include <string_view>

namespace arangodb {

class SequenceNumberTracker;  // from 05-01
class ShardWALTailer;         // from 05-02

/// REST handler for /_admin/replication/* endpoints.
/// Routes to status, start, stop, reset, feed, and checkpoint operations.
/// Follows the same envelope pattern as RestHotBackupHandler.
class RestReplicationHandler {
 public:
  enum class Route {
    STATUS,
    START,
    STOP,
    RESET,
    FEED,
    CHECKPOINT,
    UNKNOWN
  };

  RestReplicationHandler(SequenceNumberTracker& tracker,
                         ShardWALTailer& tailer);

  /// Parse route from REST path suffix.
  static Route parseRoute(std::string_view suffix);

  /// GET /_admin/replication/status
  /// Returns aggregated replication status as JSON.
  void executeStatus(std::string& response);

  /// POST /_admin/replication/start
  /// Starts replication. Returns 409 if already running.
  void executeStart(std::string& response);

  /// POST /_admin/replication/stop
  /// Stops replication gracefully. Returns 409 if already stopped.
  void executeStop(std::string& response);

  /// POST /_admin/replication/reset
  /// Resets all per-shard sequence tracking. Returns count of shards reset.
  void executeReset(std::string& response);

  /// GET /_admin/replication/feed?shardId=X&afterSequence=N&maxCount=M
  /// Streaming feed for arangosync to consume DirectMQ messages.
  void executeFeed(std::string const& shardId,
                   uint64_t afterSequence,
                   uint32_t maxCount,
                   std::string& response);

  /// POST /_admin/replication/checkpoint
  /// Body: {"shardId": "...", "appliedSequence": N}
  /// arangosync reports applied progress.
  void executeCheckpoint(std::string const& shardId,
                         uint64_t appliedSequence,
                         std::string& response);

 private:
  SequenceNumberTracker& _tracker;
  ShardWALTailer& _tailer;

  /// Format a successful response envelope.
  static std::string formatSuccess(std::string const& resultJson);

  /// Format an error response envelope.
  static std::string formatError(int code, std::string const& message);
};

}  // namespace arangodb

#endif  // ARANGODB_REST_REPLICATION_HANDLER_H
```

```cpp
#include "Enterprise/RestHandler/RestReplicationHandler.h"
#include "Enterprise/Replication/ReplicationStatusBuilder.h"
#include "Enterprise/Replication/ReplicationFeedStream.h"
#include "Enterprise/Replication/CheckpointReceiver.h"

#include <chrono>
#include <sstream>

namespace arangodb {

RestReplicationHandler::RestReplicationHandler(
    SequenceNumberTracker& tracker, ShardWALTailer& tailer)
    : _tracker(tracker), _tailer(tailer) {}

RestReplicationHandler::Route RestReplicationHandler::parseRoute(
    std::string_view suffix) {
  if (suffix == "status")     { return Route::STATUS; }
  if (suffix == "start")      { return Route::START; }
  if (suffix == "stop")       { return Route::STOP; }
  if (suffix == "reset")      { return Route::RESET; }
  if (suffix == "feed")       { return Route::FEED; }
  if (suffix == "checkpoint") { return Route::CHECKPOINT; }
  return Route::UNKNOWN;
}

void RestReplicationHandler::executeStatus(std::string& response) {
  ReplicationStatusBuilder builder(_tracker, _tailer);
  std::string statusJson = builder.build();
  response = formatSuccess(statusJson);
}

void RestReplicationHandler::executeStart(std::string& response) {
  if (_tailer.isRunning()) {
    response = formatError(409, "replication is already running");
    return;
  }
  _tailer.start();
  auto now = std::chrono::system_clock::now();
  auto epoch = std::chrono::duration_cast<std::chrono::seconds>(
      now.time_since_epoch()).count();
  std::ostringstream oss;
  oss << "{\"started\": true, \"timestamp\": " << epoch
      << ", \"previousState\": \"stopped\"}";
  response = formatSuccess(oss.str());
}

void RestReplicationHandler::executeStop(std::string& response) {
  if (!_tailer.isRunning()) {
    response = formatError(409, "replication is not running");
    return;
  }
  _tailer.stop();
  auto now = std::chrono::system_clock::now();
  auto epoch = std::chrono::duration_cast<std::chrono::seconds>(
      now.time_since_epoch()).count();
  std::ostringstream oss;
  oss << "{\"stopped\": true, \"timestamp\": " << epoch
      << ", \"previousState\": \"running\"}";
  response = formatSuccess(oss.str());
}

void RestReplicationHandler::executeReset(std::string& response) {
  auto states = _tracker.getAllStates();
  uint32_t count = static_cast<uint32_t>(states.size());
  _tracker.resetAll();
  std::ostringstream oss;
  oss << "{\"reset\": true, \"shardsReset\": " << count << "}";
  response = formatSuccess(oss.str());
}

void RestReplicationHandler::executeFeed(
    std::string const& shardId, uint64_t afterSequence,
    uint32_t maxCount, std::string& response) {
  if (shardId.empty()) {
    response = formatError(400, "missing required parameter: shardId");
    return;
  }
  if (maxCount == 0) {
    maxCount = 1000;  // default batch size
  }
  ReplicationFeedStream feed(_tailer);
  std::string feedJson = feed.fetch(shardId, afterSequence, maxCount);
  response = formatSuccess(feedJson);
}

void RestReplicationHandler::executeCheckpoint(
    std::string const& shardId, uint64_t appliedSequence,
    std::string& response) {
  if (shardId.empty()) {
    response = formatError(400, "missing required field: shardId");
    return;
  }
  if (appliedSequence == 0) {
    response = formatError(400, "appliedSequence must be greater than 0");
    return;
  }
  CheckpointReceiver receiver(_tracker);
  auto result = receiver.receive(shardId, appliedSequence);
  if (!result.accepted) {
    response = formatError(409, result.reason);
    return;
  }
  std::ostringstream oss;
  oss << "{\"acknowledged\": true, \"shardId\": \"" << shardId
      << "\", \"appliedSequence\": " << appliedSequence << "}";
  response = formatSuccess(oss.str());
}

std::string RestReplicationHandler::formatSuccess(
    std::string const& resultJson) {
  return "{\n  \"error\": false,\n  \"code\": 200,\n  \"result\": " +
         resultJson + "\n}";
}

std::string RestReplicationHandler::formatError(
    int code, std::string const& message) {
  std::ostringstream oss;
  oss << "{\n  \"error\": true,\n  \"code\": " << code
      << ",\n  \"errorMessage\": \"" << message << "\"\n}";
  return oss.str();
}

}  // namespace arangodb
```

```cpp
class SequenceNumberTracker;

/// Aggregates per-shard replication state into a JSON status object.
/// Used by RestReplicationHandler::executeStatus().

/// Output format:
/// {

  ShardWALTailer const& _tailer;
```

```cpp
#include "Enterprise/Replication/ReplicationStatusBuilder.h"

#include <algorithm>
#include <sstream>

namespace arangodb {

ReplicationStatusBuilder::ReplicationStatusBuilder(
    SequenceNumberTracker const& tracker,
    ShardWALTailer const& tailer)
    : _tracker(tracker), _tailer(tailer) {}

std::string ReplicationStatusBuilder::build() const {
  auto states = _tracker.getAllStates();
  uint64_t totalPending = _tracker.totalPending();
  uint64_t sent = _tailer.messagesSent();
  uint64_t acked = _tailer.messagesAcked();
  bool running = _tailer.isRunning();

  // Find max pending for lag estimation
  uint64_t maxPending = 0;
  for (auto const& s : states) {
    maxPending = std::max(maxPending, s.pendingCount);
  }

  double lag = estimateLagSeconds(maxPending, sent, acked);

  std::ostringstream oss;
  oss << "{\n";
  oss << "  \"isRunning\": " << (running ? "true" : "false") << ",\n";
  oss << "  \"lagSeconds\": " << lag << ",\n";
  oss << "  \"totalMessagesSent\": " << sent << ",\n";
  oss << "  \"totalMessagesAcked\": " << acked << ",\n";
  oss << "  \"totalPending\": " << totalPending << ",\n";
  oss << "  \"shardStates\": [";

  for (size_t i = 0; i < states.size(); ++i) {
    if (i > 0) {
      oss << ", ";
    }
    auto const& s = states[i];
    oss << "{\n";
    oss << "    \"shardId\": \"" << s.shardId << "\",\n";
    oss << "    \"lastGenerated\": " << s.lastGenerated << ",\n";
    oss << "    \"lastApplied\": " << s.lastApplied << ",\n";
    oss << "    \"pending\": " << s.pendingCount << "\n";
    oss << "  }";
  }

  oss << "]\n";
  oss << "}";

  return oss.str();
}

double ReplicationStatusBuilder::estimateLagSeconds(
    uint64_t maxPending, uint64_t messagesSent, uint64_t messagesAcked) {
  if (maxPending == 0) {
    return 0.0;
  }
  // Estimate throughput from acked/sent ratio. If no messages sent yet,
  // assume worst case of 1 msg/sec throughput.
  if (messagesAcked == 0 || messagesSent == 0) {
    return static_cast<double>(maxPending);
  }
  // Throughput = acked messages / (sent messages * assumed_interval)
  // Simplified: lag = maxPending * (sent / acked) as a ratio proxy
  double ratio = static_cast<double>(messagesSent) /
                 static_cast<double>(messagesAcked);
  return static_cast<double>(maxPending) * ratio;
}

}  // namespace arangodb
```

```cpp
#include "Enterprise/Replication/ReplicationFeedStream.h"

#include <sstream>

namespace arangodb {

ReplicationFeedStream::ReplicationFeedStream(ShardWALTailer const& tailer)
    : _tailer(tailer) {}

std::string ReplicationFeedStream::fetch(
    std::string const& shardId, uint64_t afterSequence,
    uint32_t maxCount) const {
  auto messages = _tailer.nextBatch(shardId, afterSequence, maxCount + 1);

  bool hasMore = messages.size() > maxCount;
  if (hasMore) {
    messages.resize(maxCount);  // trim to requested size
  }

  uint64_t lastSequence = afterSequence;

  std::ostringstream oss;
  oss << "{\n  \"messages\": [";

  for (size_t i = 0; i < messages.size(); ++i) {
    if (i > 0) {
      oss << ", ";
    }
    auto const& m = messages[i];
    oss << "{\n";
    oss << "    \"shardId\": \"" << m.shardId << "\",\n";
    oss << "    \"sequence\": " << m.sequence << ",\n";
    oss << "    \"operation\": \"" << m.operation << "\",\n";
    oss << "    \"payload\": \"" << m.payload << "\"\n";
    oss << "  }";
    lastSequence = m.sequence;
  }

  oss << "],\n";
  oss << "  \"hasMore\": " << (hasMore ? "true" : "false") << ",\n";
  oss << "  \"lastSequence\": " << lastSequence << "\n";
  oss << "}";

  return oss.str();
}
```

```cpp
#pragma once
#ifndef ARANGODB_CHECKPOINT_RECEIVER_H
#define ARANGODB_CHECKPOINT_RECEIVER_H

#include <cstdint>
#include <string>

namespace arangodb {

class SequenceNumberTracker;

/// Result of a checkpoint receive operation.
struct CheckpointResult {
  bool accepted;
  std::string reason;  // non-empty when accepted is false
};

/// Accepts checkpoint progress reports from arangosync via
/// POST /_admin/replication/checkpoint.
///
/// Validates that the reported appliedSequence is within the valid range
/// (lastApplied < appliedSequence <= lastGenerated) before updating the
/// SequenceNumberTracker.
class CheckpointReceiver {
 public:
  explicit CheckpointReceiver(SequenceNumberTracker& tracker);
```

```cpp
#include "Enterprise/Replication/CheckpointReceiver.h"

namespace arangodb {

CheckpointReceiver::CheckpointReceiver(SequenceNumberTracker& tracker)
    : _tracker(tracker) {}

CheckpointResult CheckpointReceiver::receive(
    std::string const& shardId, uint64_t appliedSequence) {
  // Look up current shard state
  auto state = _tracker.getState(shardId);

  // Validate shard exists (lastGenerated == 0 means unknown shard)
  if (state.lastGenerated == 0 && state.lastApplied == 0) {
    return CheckpointResult{false,
        "unknown shard: " + shardId};
  }

  // Validate sequence is advancing (not behind or equal to current)
  if (appliedSequence <= state.lastApplied) {
    return CheckpointResult{false,
        "appliedSequence " + std::to_string(appliedSequence) +
        " is not ahead of current lastApplied " +
        std::to_string(state.lastApplied) + " for shard " + shardId};
  }

  // Validate sequence is not ahead of what was generated
  if (appliedSequence > state.lastGenerated) {
    return CheckpointResult{false,
        "appliedSequence " + std::to_string(appliedSequence) +
        " exceeds lastGenerated " + std::to_string(state.lastGenerated) +
        " for shard " + shardId};
  }

  // Valid checkpoint -- update tracker
  _tracker.updateApplied(shardId, appliedSequence);
  return CheckpointResult{true, ""};
}

}  // namespace arangodb
```

<pitfalls>
1. **Envelope mismatch:** The success/error JSON envelope must exactly match RestHotBackupHandler's format -- same field names, same nesting. Test by comparing literal strings.
2. **JSON injection in error messages:** If shardId or error strings contain quotes, the raw ostringstream approach will produce invalid JSON. For now, shard IDs are alphanumeric, but a future hardening pass should escape special characters.
3. **Checkpoint race condition:** arangosync may send checkpoints out of order if multiple shards report concurrently. CheckpointReceiver validates per-shard independently, so cross-shard ordering is safe. But two concurrent checkpoints for the same shard could race on updateApplied -- this is acceptable for now since arangosync serializes per-shard reports.
4. **Feed stream memory:** nextBatch fetches maxCount+1 messages to detect hasMore. If maxCount is very large, this could spike memory. Enforce a ceiling (e.g., 10000) in executeFeed.
5. **Lag estimation is approximate:** estimateLagSeconds uses a throughput ratio heuristic, not wall-clock time. This is sufficient for monitoring but should not be used for correctness decisions.
6. **Missing dependency stubs:** 05-01 (SequenceNumberTracker) and 05-02 (ShardWALTailer) must be implemented or stubbed before this plan's code compiles. Tests use mocks from ReplicationMocks.h.
7. **Streaming vs. buffered response:** The current ReplicationFeedStream returns a fully buffered JSON string. True HTTP chunked streaming would require framework integration (GeneralResponse::writePayload). The buffered approach is correct for initial implementation and can be evolved to chunked transfer later.
8. **Zero appliedSequence:** CheckpointReceiver rejects appliedSequence == 0 at the handler level (executeCheckpoint), but also needs to handle the edge case where a shard's lastApplied is already 0 (fresh shard). The current logic treats lastGenerated == 0 && lastApplied == 0 as "unknown shard" which is correct for fresh state.
</pitfalls>
